import {Component} from '@angular/core'; import {CdkDragDrop, moveItemInArray, CdkDrag, CdkDropList} from '@angular/cdk/drag-drop'; /** * @title Drag&Drop sort predicate */ @Component({ selector: 'cdk-drag-drop-sort-predicate-example', templateUrl: 'cdk-drag-drop-sort-predicate-example.html', styleUrl: 'cdk-drag-drop-sort-predicate-example.css', imports: [CdkDropList, CdkDrag], }) export class CdkDragDropSortPredicateExample { numbers = [1, 2, 3, 4, 5, 6, 7, 8]; drop(event: CdkDragDrop<unknown>) { moveItemInArray(this.numbers, event.previousIndex, event.currentIndex); } /** * Predicate function that only allows even numbers to be * sorted into even indices and odd numbers at odd indices. */ sortPredicate(index: number, item: CdkDrag<number>) { return (index + 1) % 2 === item.data % 2; } }
